core.Choose    (function)  
    The 'Choose' function returns a result from 'choices' based on the 'choice' argument. The 'choices' argument is an object, where the members are the choices. When 'choice' matches the name of one of the members of 'choices', it will return that members' value.

File  
    pops/pops.core.js

Syntax  
var foo = Choose(choice, choices);
  • choice : A string value representing your choice.
  • choices : An object containing the choice values. Each member is a choice, it's name is the choice's name. You can use multiple choice names for a single choice using the 'or' operator (_?_).

Choices  
  • $else: The value of this member will be returned if no choices can be matched.  
    $else: val,
    • val : Any value.
  • $run: If set to true, and the resulting value is a function, Choose will run the function and return the result rather than returning the function itself.  
    $run: true,
Naming syntax  
    Naming a choice is like naming a variable, the same rules apply.

Name     Result    
myChoicegood
my Choiceerror
myChoice2good
2myChoiceerror
my_Choicegood
$myChoicegood
my@Choiceerror
?myChoiceerror

_or_  
    Use  _or_  to set a single value to multiple choices.
Choice1_or_Choice2_or_MyChoice: val,
  • Choice1, Choice2, MyChoice : These are choice names.
  • val : Any value.

Examples  
Choose  
require('pops/pops.loader').Load();

var choice = 'choice2';

var result = Choose(choice, {
    choice1: '- Choice 1 -',
    choice2: '- Choice 2 -',
    choice3: '- Choice 3 -',
    choice4: '- Choice 4 -'
});

cout(result);


$else  
require('pops/pops.loader').Load();

var choices = {
    choice1: '- Choice 1 -',
    choice2: '- Choice 2 -',
    $else: '- Else -',
};

cout(Choose('choice2', choices));
cout(Choose('choice6', choices));
cout(Choose('choice1', choices));


$run  
require('pops/pops.loader').Load();

var choices = {
    foo: function() { return 'FOO!'; },
    Bar: 'BAR!',
    hELLo: function() { return 'HELLO!'; },
    world: function() { return function() { return 'WORLD!'; } },
};

var choices2 = {
    $run: true,
    foo: function() { return 'FOO!'; },
    Bar: 'BAR!',
    hELLo: function() { return 'HELLO!'; },
    world: function() { return function() { return 'WORLD!'; } },
};

cout(Choose('Foo', choices));
cout(Choose('Bar', choices));
cout(Choose('hello', choices));
cout(Choose('wORLd', choices));

cout();

cout(Choose('Foo', choices2));
cout(Choose('Bar', choices2));
cout(Choose('hello', choices2));
cout(Choose('wORLd', choices2));


_or_  
require('pops/pops.loader').Load();

var choices = {
    choice1: '- Choice 1 -',
    choice2_or_choice3: '- Choice 2 & 3 -',
    choice4: '- Choice 4 -',
};

cout(Choose('choice1', choices));
cout(Choose('choice2', choices));
cout(Choose('choice3', choices));
cout(Choose('choice4', choices));
