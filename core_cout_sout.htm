core.cout    (function)  
    Writes a string to the console. cout moves the cursor to the next line when it is finished witing.

core.sout    (function)  
    Same as cout, except it does not move the cursor to the next line.

File  
    pops/pops.core.js

Syntax  
cout(str);
sout(str);
  • str : Any string value.

Examples  
cout & sout  
require('pops/pops.core').Global.Pops();

var v1;

cout('cout \n----------------');
for(v1 = 1; v1 < 9; v1++) {
    cout(v1);
}

cout('sout \n----------------');
for(v1 = 1; v1 < 9; v1++) {
    sout(v1);
}
